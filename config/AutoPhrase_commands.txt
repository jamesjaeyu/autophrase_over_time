For running on Windows:

winpty docker run -v "/${PWD}/data":/autophrase/data -v "/${PWD}/models":/autophrase/models -it \
    -e RAW_TRAIN=data/dblp-v13/content_YEAR.txt \
    -e ENABLE_POS_TAGGING=1 \
    -e MIN_SUP=30 -e THREAD=10 \
    -e MODEL=models/content_YEAR \
    -e TEXT_TO_SEG=data/dblp-v13/content_YEAR.txt \
    remenberl/autophrase

./auto_phrase.sh
./phrasal_segmentation.sh
